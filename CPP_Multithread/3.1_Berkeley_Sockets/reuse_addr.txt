int optval = 1;
setsockopt(sd, SOL_SOCKET, SO_REUSEADDR, &optval, sizeof(optval));

struct timeval tv;
tv.tv_sec = 16; // seconds
tv.tv_usec = 0; // misec

Timeout to send
setsockopt(sd, SOL_SOCKET, SO_RCVTIMEO, (char *)&tv, sizeof(tv));

Timeout to recieve
setsockopt(sd, SOL_SOCKET, SO_SNDTIMEO, (char *)&tv, sizeof(tv));

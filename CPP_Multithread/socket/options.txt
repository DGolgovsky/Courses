// Reuse address after killing app
int optval = 1
setsockopt(s, SOL_SOCKET, SO_REUSEADDR, &optval, sizeof(optval));

// timeout
struct timeval tv;
    tv.tv_sec = 16;
    tv_tv_usec = 0;
recv: setsockopt(s, SO_SOCKET, SO_RCVTIMEO, (char *) &tv, sizeof(tv));
send: setsockopt(s, SO_SOCKET, SO_SNDTIMEO, (char *) &tv, sizeof(tv));
